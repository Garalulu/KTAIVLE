```python
def analys_c(var, target, data):
    sns.barplot(x=var, y=target, data = data)
    plt.grid()
    plt.show()
    
    d_array = []
    for i in data[var].unique():
        d_array.append(data.loc[data[var]==i, target])
    if len(d_array) == 2:
        print(spst.ttest_ind(d_array[0], d_array[1]))
    print(spst.f_oneway(*d_array))
```
